SELECT
  CAREER_LENGTHS.NAME,
  CAREER_LENGTHS.END_YEAR - CAREER_LENGTHS.START_YEAR AS LONGEVITY
FROM (
  SELECT
    NAME,
    MIN(YEAR) AS START_YEAR,
    MAX(YEAR) AS END_YEAR
  FROM
    ACTORS
    JOIN MOVIES ON ACTORS.MID = MOVIES.MID
  GROUP BY
    NAME) AS CAREER_LENGTHS
ORDER BY
  LONGEVITY DESC
LIMIT 1;
